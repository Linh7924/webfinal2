Bài 1: Tính và in tổng các số nguyên tố từ 1 đến 100


// Hàm kiểm tra số nguyên tố
function isPrime($num) {
    if ($num < 2) return false;
    for ($i = 2; $i <= sqrt($num); $i++) {
        if ($num % $i == 0) return false;
    }
    return true;
}

// Tính tổng số nguyên tố từ 1 đến 100
$sum = 0;
for ($i = 1; $i <= 100; $i++) {
    if (isPrime($i)) {
        $sum += $i;
    }
}
echo "Tổng các số nguyên tố từ 1 đến 100 là: " . $sum;

Bài 2: Tính tổng chuỗi

a. T = 1/2 + 2/3 + 3/4 + … n/n+1


function calculateSeriesA($n) {
    $sum = 0;
    for ($i = 1; $i <= $n; $i++) {
        $sum += $i / ($i + 1);
    }
    return $sum;
}

$n = 10; // Giá trị n có thể thay đổi
$resultA = calculateSeriesA($n);
echo "Tổng chuỗi a với n = $n là: " . round($resultA, 4);

b. T = 1/2 + 1/4 + 1/6 + …. 1/n+2 với e > 0.0001


function calculateSeriesB() {
    $sum = 0;
    $n = 2;
    $term = 1 / $n;
    
    while ($term > 0.0001) {
        $sum += $term;
        $n += 2;
        $term = 1 / $n;
    }
    
    return array('sum' => $sum, 'terms' => $n / 2 - 1);
}

$resultB = calculateSeriesB();
echo "Tổng chuỗi b là: " . round($resultB['sum'], 4);
echo "<br>Số phần tử: " . $resultB['terms'];
